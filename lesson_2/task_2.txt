# Создать файл с наполнением, используя несколько способов.
# Использовать разобранные текстовые редакторы для наполнения файлов данными.
# Создать копии созданных файлов, создать несколько каталогов с подкаталогами, перенести несколько файлов в созданные каталоги.
# Перечислить команды и используемые параметры команд.

# Способы создания файла
touch file1.txt # создаёт пустой файл
vim file2.txt # создаёт и открывает пустой файл в vim
nano file3.txt # создаёт и открывает пустой файл в nano

-----------------Vim-------------------
# Откроем файл с помощью редактора Vim
vim file1.txt

# Перейдем в режим редактирования путём нажатия i

# Внесём изменения и выйдем из режима редактирования путём нажатия Esc

# Перейдём в режим последней строки путём нажатия клавиши ":"

# Ввёдем одну из доступных команд
q # выйти из редактора
q! # выйти из редактора без сохранения изменений
w # сохранить файл
wq # сохранить и выйти из редактора
---------------------------------------

-----------------Nano------------------
# Откроем файл с помощью редактора Nano
nano file2.txt

# Внесём изменения и выйдем из редактора путём нажатия комбинации клавиш Ctrl + X

# При запросе о сохранении изменений выберем подходящий вариант(сохранить или нет)
---------------------------------------

# Копируем file1.txt в file2.txt
cp file1.txt file2.txt

# Создать каталог
mkdir catalog1

# Создать подкаталог созданного каталога
cd catalog1 # переместились в нужный каталог
mkdir catalog1.1 # создали подкаталог catalog1.1 в каталоге catalog1

# Переместить файл в подкаталог
mv file1.txt catalog1/catalog1.1
