# Создать два произвольных файла.
touch file1.txt
touch file2.txt

# Первому присвоить права на чтение и запись для владельца и группы, только на чтение — для всех.
chmod ug+rw file1.txt # первый шаг символьного вида
сhmod o+r file1.txt # второй шаг символьного вида
chmod 664 file1.txt # численный вид

# Второму присвоить права на чтение и запись только для владельца. Сделать это в численном и символьном виде.
chmod u+rw file2.txt # символьный вид
chmod 600 file2.txt # численный вид
